#include<stdio.h>
#include<string.h>

int superDigitRec(long long num) {
    if(num < 10) {
        return (int) num;
    }
    long long sum = 0;
    while(num > 0)  {
        sum += num % 10;
        num /= 10;
    }
    return superDigitRec(sum);
}
int superDigit(char* n, int k) {
    long long sum = 0;
    
    for(int i = 0; n[i] != '\0'; i++) {
        sum += (n[i] - '0');
    }
    sum *= k;
    return superDigitRec(sum);
}
int main() {
    char n[100001];
    int k;
    scanf("%s %d", n, &k);
    int result = superDigit(n,k);
    printf("%d\n", result);
    
    return 0;
}
