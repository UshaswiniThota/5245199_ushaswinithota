#include<stdio.h>
int birthday(int s[], int n, int d, int m) {
    int count = 0, sum = 0;
    for(int i = 0; i < n; i++) {
        sum += s[i];
        if(i >= m) sum -= s[i - m];
        if(i >= m - 1 && sum == d) count++;
    }
    return count;
}
int main() {
    int n, d, m;
    scanf("%d", &n);
    int s[n];
    for(int i = 0; i < n; i++) scanf("%d", &s[i]);
    scanf("%d %d", &d, &m);
    printf("%d\n", birthday(s, n, d, m));
    return 0;
}
