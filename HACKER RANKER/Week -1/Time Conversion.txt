#include<stdio.h>
#include<stdlib.h>
#include<string.h>

char* timeConversion(char* s) {
    char* result = (char*)malloc(9 * sizeof(char));
    int hour;
    char period[3];
    
    sscanf(s, "%2d:%*2d:%*2d%2s", &hour, period);
    if (strcmp(period, "AM") == 0) {
        if (hour == 12) hour = 0;
    } else {
        if (hour != 12) hour += 12;
    }
    sprintf(result, "%02d:%c%c:%c%c", hour, s[3], s[4], s[6], s[7]);
    return result;
}
int main() 
{
    char s[11];
    scanf("%s", s);
    
    char* result = timeConversion(s);
    printf("%s\n", result);
    free(result);
    
    return 0;
}
