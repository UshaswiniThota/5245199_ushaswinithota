#include<stdio.h>
#include<stdlib.h>

int sockMerchant(int n, int ar[]) {
    int max_color = 0;
    for (int i = 0; i< n; i++ ) {
        if(ar[i] > max_color) {
            max_color = ar[i];
        }
    }  
    
    int *counts = (int *)calloc(max_color + 1, sizeof(int));
    if(counts == NULL) {
        return -1;
    }
    for(int i = 0; i<n; i++) {
        counts[ar[i]]++;
    }
    int pairs = 0;
    for(int i=0;i<= max_color;i++)  {
        pairs += counts[i] / 2;
    }
    free(counts);
    return pairs;
}
int main() {
    int n;
    scanf("%d", &n);
    
    int *ar = (int *)malloc(n * sizeof(int));
    for(int i = 0;i<n;i++) {
        scanf("%d", &ar[i]);
    }
    int result = sockMerchant(n, ar);
    printf("%d\n", result);
    
    free(ar);
    return 0;
}
