#include<stdio.h>
#include<ctype.h>

char* caesarCipher(char* s, int k) {
    static char result[1001];
    int i =0;
    k = k % 26;
    while (s[i] != '\0') {
        char c = s[i];
        if(isalpha(c)) {
            if(islower(c)) {
                result[i] = ((c - 'a' + k) % 26) + 'a';
            }
            else{
                result[i] = ((c - 'A' + k) % 26) + 'A';
            }
        }else{
            result[i] = c;
        }
        i++;
    }
    result[i] = '\0';
    return result;
}
int main() {
    int n, k;
    scanf("%d", &n);
    char s[1001];
    scanf("%s", s);
    scanf("%d",&k);
    
    printf("%s\n", caesarCipher(s,k));
    return 0;
}
