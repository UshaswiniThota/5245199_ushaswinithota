#include<stdio.h>
int pageCount(int n, int p) {
    int front_turns = p / 2;
    int back_truns = (n / 2) - (p / 2);
    return (front_turns <back_truns) ? front_turns : back_truns;
}
int main() {
    int n, p;
    scanf("%d", &n);
    scanf("%d", &p);
    printf("%d\n", pageCount(n, p));
    return 0;
}
