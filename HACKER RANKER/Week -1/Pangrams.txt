#include<stdio.h>
#include<ctype.h>
#include<string.h>
#include<stdbool.h>

char* pangrams(char* s) {
    bool letters[26] = {false};
    
    for(int i = 0; s[i] != '\0'; i++) {
        if (isalpha(s[i])) {
            char lower = tolower(s[i]);
            letters[lower - 'a'] = true;
        }
    }
    for( int i = 0; i < 26; i++) {
        if (!letters[i]) {
            return "not pangram";
        }
    }
    return "pangram";
}
int main() {
    char s[1001];
    fgets(s, sizeof(s), stdin);
    
    s[strcspn(s, "\n")] = '\0';
    printf("%s\n", pangrams(s));
    return 0;
}
